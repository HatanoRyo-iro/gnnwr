 188    def init_optimizer(self, optimizer, optimizer_params=None):
 189        r"""
 190        Initialize the optimizer.
 191
 192        This method sets up the optimizer and the learning rate scheduler with the given parameters.
 193
 194        Parameters
 195        ----------
 196        optimizer : str
 197            The name of the optimizer to use. This should be one of the supported optimizers.
 198        optimizer_params : dict, optional
 199            A dictionary containing parameters for the optimizer and scheduler. If not provided,
 200            default values will be used. The dictionary can contain the following keys:
 201
 202            - **scheduler** \: *str*
 203            
 204            The type of learning rate scheduler to use. Valid options include ``Special``,
 205            ``Constant``, and ``MultiStepLR``.
 206            - **maxlr** \: *float*
 207                
 208                The maximum learning rate for the scheduler, for ``Special``.
 209            - **minlr** \: *float*
 210                
 211                The minimum learning rate for the scheduler, for ``Special``.
 212            - **upepoch** \: *int*
 213                
 214                The number of epochs until the maximum learning rate is reached, for ``Special``.
 215            - **decayepoch** \: *int*
 216                
 217                The epoch at which learning rate decay starts, for ``Special``.
 218            - **decayrate** \: *float*
 219                
 220                The rate at which the learning rate decays, for ``Special``.
 221            - **stop_change_epoch** \: *int*
 222
 223                The epoch at which to stop adjusting the learning rate, for ``Special``.
 224            - **stop_lr** \: *float*
 225
 226                The learning rate to stop at after the specified epoch, for ``Special``.
 227            - **scheduler_milestones** \: *list*
 228
 229                The epochs at which to decay the learning rate for a ``MultiStepLR`` scheduler.
 230            - **scheduler_gamma** \: *float*
 231
 232                The factor by which the learning rate is reduced for a ``MultiStepLR`` scheduler.
 233            - **weight_decay** \: *float*
 234
 235                The weight decay factor for the optimizer.
 236
 237        Raises
 238        ------
 239        ValueError
 240            If an unsupported optimizer or scheduler is specified.
 241
 242        """
 243        # initialize the optimizer
 244        if optimizer_params is None:
 245            optimizer_params = {}
 246        weigth_decay = optimizer_params.get("weight_decay", 1e-3)
 247        
 248        if optimizer == "SGD":
 249            self._optimizer = optim.SGD(
 250                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
 251        elif optimizer == "Adam":
 252            self._optimizer = optim.Adam(
 253                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
 254        elif optimizer == "AdamW":
 255            self._optimizer = optim.AdamW(
 256                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
 257        elif optimizer == "RMSprop":
 258            self._optimizer = optim.RMSprop(
 259                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
 260        elif optimizer == "Adagrad":
 261            self._optimizer = optim.Adagrad(
 262                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
 263        elif optimizer == "Adadelta":
 264            self._optimizer = optim.Adadelta(
 265                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
 266        else:
 267            raise ValueError("Invalid Optimizer")
 268        self._optimizer_name = optimizer  # optimizer name
 269
 270        # lr scheduler
 271        
 272        scheduler = optimizer_params.get("scheduler", "Constant")
 273        scheduler_milestones = optimizer_params.get(
 274            "scheduler_milestones", [100, 500, 1000, 2000])
 275        scheduler_gamma = optimizer_params.get("scheduler_gamma", 0.5)
 276        scheduler_T_max = optimizer_params.get("scheduler_T_max", 1000)
 277        scheduler_eta_min = optimizer_params.get("scheduler_eta_min", 0.01)
 278        scheduler_T_0 = optimizer_params.get("scheduler_T_0", 100)
 279        scheduler_T_mult = optimizer_params.get("scheduler_T_mult", 3)
 280        if scheduler == "MultiStepLR":
 281            self._scheduler = optim.lr_scheduler.MultiStepLR(
 282                self._optimizer, milestones=scheduler_milestones, gamma=scheduler_gamma)
 283        elif scheduler == "CosineAnnealingLR":
 284            self._scheduler = optim.lr_scheduler.CosineAnnealingLR(
 285                self._optimizer, T_max=scheduler_T_max, eta_min=scheduler_eta_min)
 286        elif scheduler == "CosineAnnealingWarmRestarts":
 287            self._scheduler = optim.lr_scheduler.CosineAnnealingWarmRestarts(
 288                self._optimizer, T_0=scheduler_T_0, T_mult=scheduler_T_mult, eta_min=scheduler_eta_min)
 289        elif scheduler == "Special":
 290            maxlr = optimizer_params.get("maxlr", 0.1)
 291            minlr = optimizer_params.get("minlr", 0.01)
 292            upepoch = optimizer_params.get("upepoch", 10000)
 293            uprate = (maxlr - minlr) / upepoch
 294            decayepoch = optimizer_params.get("decayepoch", 20000)
 295            decayrate = optimizer_params.get("decayrate", 0.95)
 296            stop_change_epoch = optimizer_params.get("stop_change_epoch", 30000)
 297            stop_lr = optimizer_params.get("stop_lr", 0.001)
 298            lambda_lr = lambda epoch: epoch * uprate + minlr if epoch < upepoch else (
 299                maxlr if epoch < decayepoch else maxlr * (
 300                        decayrate ** ((epoch - decayepoch) // 200))) if epoch < stop_change_epoch else stop_lr
 301            self._scheduler = optim.lr_scheduler.LambdaLR(
 302                self._optimizer, lr_lambda=lambda_lr)
 303        elif scheduler == "Constant":
 304            self._scheduler = optim.lr_scheduler.LambdaLR(
 305                self._optimizer, lr_lambda=lambda epoch: 1)
 306        else:
 307            raise ValueError("Invalid Scheduler")

 309    def __train(self):
 310        """
 311        train the network
 312        """
 313        self._model.train()  # set the model to train mode
 314        train_loss = 0  # initialize the loss
 315        data_loader = self._train_dataset.dataloader  # get the data loader
 316        weight_all = torch.tensor([],dtype=torch.float32,device=self._device)
 317        x_true = torch.tensor([],dtype=torch.float32,device=self._device)
 318        y_true = torch.tensor([],dtype=torch.float32,device=self._device)
 319        y_pred = torch.tensor([],dtype=torch.float32,device=self._device)
 320        for index, (data, coef, label, data_index) in enumerate(data_loader):
 321            # move the data to gpu
 322            data, coef, label = data.to(self._device), coef.to(self._device), label.to(self._device)
 323
 324            self._optimizer.zero_grad()  # zero the gradient
 325            if self._optimizer_name == "Adagrad":
 326                # move optimizer state to gpu
 327                for state in self._optimizer.state.values():
 328                    for k, v in state.items():
 329                        if isinstance(v, torch.Tensor):
 330                            state[k] = v.to(self._device)
 331
 332            x_true = torch.cat((x_true, coef), 0)
 333            y_true = torch.cat((y_true, label), 0)
 334            weight = self._model(data)
 335
 336            weight_all = torch.cat((weight_all, weight.to(torch.float32)), 0)
 337            output = self._out(weight.mul(coef))
 338            y_pred = torch.cat((y_pred, output), 0)
 339            loss = self._criterion(output, label) # calculate the loss
 340            loss.backward()  # back propagation
 341            self._optimizer.step()  # update the parameters
 342            if isinstance(data, list):
 343                train_loss += loss.item() * data[0].size(0)
 344            else:
 345                train_loss += loss.item() * data.size(0)  # accumulate the loss
 346
 347        self._train_diagnosis = DIAGNOSIS(weight_all, x_true, y_true, y_pred)
 348        train_loss /= self._train_dataset.datasize  # calculate the average loss
 349        self._trainLossList.append(train_loss)  # record the loss

 351    def __valid(self):
 352        """
 353        validate the network
 354        """
 355        self._model.eval()  # set the model to validation mode
 356        val_loss = 0  # initialize the loss
 357        label_list = torch.tensor([],dtype=torch.float32,device=self._device)
 358        out_list = torch.tensor([],dtype=torch.float32,device=self._device)
 359
 360        data_loader = self._valid_dataset.dataloader  # get the data loader
 361
 362        with torch.no_grad():  # disable gradient calculation
 363            for data, coef, label, data_index in data_loader:
 364                data, coef, label = data.to(self._device), coef.to(self._device), label.to(self._device)
 365                # weight = self._model(data)
 366                output = self._out(self._model(
 367                    data).mul(coef.to(torch.float32)))
 368                loss = self._criterion(output, label)  # calculate the loss
 369                out_list = torch.cat((out_list, output.view(-1)))
 370                label_list = torch.cat((label_list, label.view(-1)))
 371                if isinstance(data, list):
 372                    val_loss += loss.item() * data[0].size(0)
 373                else:
 374                    val_loss += loss.item() * data.size(0)  # accumulate the loss
 375            val_loss /= len(self._valid_dataset)  # calculate the average loss
 376            self._validLossList.append(val_loss)  # record the loss
 377            try:
 378                r2 =1 - torch.sum((out_list - label_list) ** 2) / torch.sum((label_list - torch.mean(label_list)) ** 2)
 379            except:
 380                if np.isnan(out_list).sum() > 0:
 381                    raise ValueError("The output contains nan value")
 382                else:
 383                    raise ValueError("The Unexpected Error")
 384            self._valid_r2 = r2
 385            if r2 > self._bestr2:
 386                # if the R square is better than the best R square,record the R square and save the model
 387                self._bestr2 = r2
 388                self._besttrainr2 = self._train_diagnosis.R2().data
 389                self._noUpdateEpoch = 0
 390                if not os.path.exists(self._modelSavePath):
 391                    os.mkdir(self._modelSavePath)
 392                torch.save(self._model, self._modelSavePath + '/' + self._modelName + ".pkl")
 393            else:
 394                self._noUpdateEpoch += 1

189    def init_optimizer(self, optimizer, optimizer_params=None):
190        r"""
191        Initialize the optimizer.
192
193        This method sets up the optimizer and the learning rate scheduler with the given parameters.
194
195        Parameters
196        ----------
197        optimizer : str
198            The name of the optimizer to use. This should be one of the supported optimizers.
199        optimizer_params : dict, optional
200            A dictionary containing parameters for the optimizer and scheduler. If not provided,
201            default values will be used. The dictionary can contain the following keys:
202
203            - **scheduler** \: *str*
204            
205            The type of learning rate scheduler to use. Valid options include ``Special``,
206            ``Constant``, and ``MultiStepLR``.
207            - **maxlr** \: *float*
208                
209                The maximum learning rate for the scheduler, for ``Special``.
210            - **minlr** \: *float*
211                
212                The minimum learning rate for the scheduler, for ``Special``.
213            - **upepoch** \: *int*
214                
215                The number of epochs until the maximum learning rate is reached, for ``Special``.
216            - **decayepoch** \: *int*
217                
218                The epoch at which learning rate decay starts, for ``Special``.
219            - **decayrate** \: *float*
220                
221                The rate at which the learning rate decays, for ``Special``.
222            - **stop_change_epoch** \: *int*
223
224                The epoch at which to stop adjusting the learning rate, for ``Special``.
225            - **stop_lr** \: *float*
226
227                The learning rate to stop at after the specified epoch, for ``Special``.
228            - **scheduler_milestones** \: *list*
229
230                The epochs at which to decay the learning rate for a ``MultiStepLR`` scheduler.
231            - **scheduler_gamma** \: *float*
232
233                The factor by which the learning rate is reduced for a ``MultiStepLR`` scheduler.
234            - **weight_decay** \: *float*
235
236                The weight decay factor for the optimizer.
237
238        Raises
239        ------
240        ValueError
241            If an unsupported optimizer or scheduler is specified.
242
243        """
244        # initialize the optimizer
245        if optimizer_params is None:
246            optimizer_params = {}
247        weigth_decay = optimizer_params.get("weight_decay", 1e-3)
248        
249        if optimizer == "SGD":
250            self._optimizer = optim.SGD(
251                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
252        elif optimizer == "Adam":
253            self._optimizer = optim.Adam(
254                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
255        elif optimizer == "AdamW":
256            self._optimizer = optim.AdamW(
257                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
258        elif optimizer == "RMSprop":
259            self._optimizer = optim.RMSprop(
260                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
261        elif optimizer == "Adagrad":
262            self._optimizer = optim.Adagrad(
263                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
264        elif optimizer == "Adadelta":
265            self._optimizer = optim.Adadelta(
266                self._model.parameters(), lr=self._start_lr, weight_decay=weigth_decay)
267        else:
268            raise ValueError("Invalid Optimizer")
269        self._optimizer_name = optimizer  # optimizer name
270
271        # lr scheduler
272        
273        scheduler = optimizer_params.get("scheduler", "Constant")
274        scheduler_milestones = optimizer_params.get(
275            "scheduler_milestones", [100, 500, 1000, 2000])
276        scheduler_gamma = optimizer_params.get("scheduler_gamma", 0.5)
277        scheduler_T_max = optimizer_params.get("scheduler_T_max", 1000)
278        scheduler_eta_min = optimizer_params.get("scheduler_eta_min", 0.01)
279        scheduler_T_0 = optimizer_params.get("scheduler_T_0", 100)
280        scheduler_T_mult = optimizer_params.get("scheduler_T_mult", 3)
281        if scheduler == "MultiStepLR":
282            self._scheduler = optim.lr_scheduler.MultiStepLR(
283                self._optimizer, milestones=scheduler_milestones, gamma=scheduler_gamma)
284        elif scheduler == "CosineAnnealingLR":
285            self._scheduler = optim.lr_scheduler.CosineAnnealingLR(
286                self._optimizer, T_max=scheduler_T_max, eta_min=scheduler_eta_min)
287        elif scheduler == "CosineAnnealingWarmRestarts":
288            self._scheduler = optim.lr_scheduler.CosineAnnealingWarmRestarts(
289                self._optimizer, T_0=scheduler_T_0, T_mult=scheduler_T_mult, eta_min=scheduler_eta_min)
290        elif scheduler == "Special":
291            maxlr = optimizer_params.get("maxlr", 0.1)
292            minlr = optimizer_params.get("minlr", 0.01)
293            upepoch = optimizer_params.get("upepoch", 10000)
294            uprate = (maxlr - minlr) / upepoch
295            decayepoch = optimizer_params.get("decayepoch", 20000)
296            decayrate = optimizer_params.get("decayrate", 0.95)
297            stop_change_epoch = optimizer_params.get("stop_change_epoch", 30000)
298            stop_lr = optimizer_params.get("stop_lr", 0.001)
299            lambda_lr = lambda epoch: epoch * uprate + minlr if epoch < upepoch else (
300                maxlr if epoch < decayepoch else maxlr * (
301                        decayrate ** ((epoch - decayepoch) // 200))) if epoch < stop_change_epoch else stop_lr
302            self._scheduler = optim.lr_scheduler.LambdaLR(
303                self._optimizer, lr_lambda=lambda_lr)
304        elif scheduler == "Constant":
305            self._scheduler = optim.lr_scheduler.LambdaLR(
306                self._optimizer, lr_lambda=lambda epoch: 1)
307        else:
308            raise ValueError("Invalid Scheduler")

310    def __train(self):
311        """
312        train the network
313        """
314        self._model.train()  # set the model to train mode
315        train_loss = 0  # initialize the loss
316        data_loader = self._train_dataset.dataloader  # get the data loader
317        weight_all = torch.tensor([],dtype=torch.float32,device=self._device)
318        x_true = torch.tensor([],dtype=torch.float32,device=self._device)
319        y_true = torch.tensor([],dtype=torch.float32,device=self._device)
320        y_pred = torch.tensor([],dtype=torch.float32,device=self._device)
321        for index, (data, coef, label, data_index) in enumerate(data_loader):
322            # move the data to gpu
323            data, coef, label = data.to(self._device), coef.to(self._device), label.to(self._device)
324
325            self._optimizer.zero_grad()  # zero the gradient
326            if self._optimizer_name == "Adagrad":
327                # move optimizer state to gpu
328                for state in self._optimizer.state.values():
329                    for k, v in state.items():
330                        if isinstance(v, torch.Tensor):
331                            state[k] = v.to(self._device)
332
333            x_true = torch.cat((x_true, coef), 0)
334            y_true = torch.cat((y_true, label), 0)
335            weight = self._model(data)
336
337            weight_all = torch.cat((weight_all, weight.to(torch.float32)), 0)
338            output = self._out(weight.mul(coef))
339            y_pred = torch.cat((y_pred, output), 0)
340            loss = self._criterion(output, label) # calculate the loss
341            loss.backward()  # back propagation
342            self._optimizer.step()  # update the parameters
343            if isinstance(data, list):
344                train_loss += loss.item() * data[0].size(0)
345            else:
346                train_loss += loss.item() * data.size(0)  # accumulate the loss
347
348        self._train_diagnosis = DIAGNOSIS(weight_all, x_true, y_true, y_pred)
349        train_loss /= self._train_dataset.datasize  # calculate the average loss
350        self._trainLossList.append(train_loss)  # record the loss

352    def __valid(self):
353        """
354        validate the network
355        """
356        self._model.eval()  # set the model to validation mode
357        val_loss = 0  # initialize the loss
358        label_list = torch.tensor([],dtype=torch.float32,device=self._device)
359        out_list = torch.tensor([],dtype=torch.float32,device=self._device)
360
361        data_loader = self._valid_dataset.dataloader  # get the data loader
362
363        with torch.no_grad():  # disable gradient calculation
364            for data, coef, label, data_index in data_loader:
365                data, coef, label = data.to(self._device), coef.to(self._device), label.to(self._device)
366                # weight = self._model(data)
367                output = self._out(self._model(
368                    data).mul(coef.to(torch.float32)))
369                loss = self._criterion(output, label)  # calculate the loss
370                out_list = torch.cat((out_list, output.view(-1)))
371                label_list = torch.cat((label_list, label.view(-1)))
372                if isinstance(data, list):
373                    val_loss += loss.item() * data[0].size(0)
374                else:
375                    val_loss += loss.item() * data.size(0)  # accumulate the loss
376            val_loss /= len(self._valid_dataset)  # calculate the average loss
377            self._validLossList.append(val_loss)  # record the loss
378            try:
379                r2 =1 - torch.sum((out_list - label_list) ** 2) / torch.sum((label_list - torch.mean(label_list)) ** 2)
380            except:
381                if np.isnan(out_list).sum() > 0:
382                    raise ValueError("The output contains nan value")
383                else:
384                    raise ValueError("The Unexpected Error")
385            self._valid_r2 = r2
386            if r2 > self._bestr2:
387                # if the R square is better than the best R square,record the R square and save the model
388                self._bestr2 = r2
389                self._besttrainr2 = self._train_diagnosis.R2().data
390                self._noUpdateEpoch = 0
391                if not os.path.exists(self._modelSavePath):
392                    os.mkdir(self._modelSavePath)
393                torch.save(self._model, self._modelSavePath + '/' + self._modelName + ".pkl")
394            else:
395                self._noUpdateEpoch += 1

433    def run(self, max_epoch=1, early_stop=-1,**kwargs):
434        """
435        train the model and validate the model
436
437        Parameters
438        ----------
439        max_epoch : int
440            the max epoch of the training (default: ``1``)
441        early_stop : int
442            if the model has not been updated for ``early_stop`` epochs, the training will stop (default: ``-1``)

444            if ``early_stop`` is ``-1``, the training will not stop until the max epoch
445        print_frequency : int
446            the frequency of printing the information (default: ``50``)
447
448        show_detailed_info : bool
449            if ``True``, the detailed information will be shown (default: ``True``)
450        """
451        if kwargs.get("print_frequency") is not None:
452            warnings.warn("The parameter print_frequency is deprecated, the information will be shown in tqdm")
453        if kwargs.get("show_detailed_info") is not None:
454            warnings.warn("The parameter show_detailed_info is deprecated, the information will be shown in tqdm")
455        # model selection method
456        model_selection = kwargs.get("model_selection", "val")
457        self.__istrained = True
458        if self._use_gpu:
459            self._model = nn.DataParallel(module=self._model)  # parallel computing
460            self._model = self._model.cuda()
461            self._out = self._out.cuda()
462        # create file
463        if not os.path.exists(self._log_path):
464            os.mkdir(self._log_path)
465        file_str = self._log_path + self._log_file_name
466        logging.basicConfig(format='%(asctime)s - %(filename)s[line:%(lineno)d] - %(levelname)s: %(message)s',
467                            filename=file_str, level=logging.INFO)
468        with tqdm(range(max_epoch)) as pbar:
469            for epoch in pbar:
470                self._epoch = epoch
471                # train the network
472                # record the information of the training process
473                self.__train()
474                # validate the network
475                # record the information of the validation process
476                self.__valid()
477                # out put the information
478                pbar.set_postfix({'Train Loss': "{:5f}".format(self._trainLossList[-1]), 'Train R2': "{:5f}".format(self._train_diagnosis.R2().data.cpu().numpy()),
479                                  'Train AIC': self._train_diagnosis.AIC(),'Valid Loss': self._validLossList[-1],
480                                  'Valid R2': self._valid_r2.item(), 'Best Valid R2': self._bestr2.item(),
481                                  'Learning Rate': self._optimizer.param_groups[0]['lr']})
482
483                self._scheduler.step()  # update the learning rate
484                # tensorboard

486                self._writer.add_scalar('Training/Loss', self._trainLossList[-1], self._epoch)

488                self._writer.add_scalar('Training/RMSE', self._train_diagnosis.RMSE().data, self._epoch)
489                self._writer.add_scalar('Training/AIC', self._train_diagnosis.AIC(), self._epoch)
490                self._writer.add_scalar('Training/AICc', self._train_diagnosis.AICc(), self._epoch)
491                self._writer.add_scalar('Validation/Loss', self._validLossList[-1], self._epoch)
492                self._writer.add_scalar('Validation/R2', self._valid_r2, self._epoch)
493                self._writer.add_scalar('Validation/Best R2', self._bestr2, self._epoch)
494
495                # log output
496                log_str = "Epoch: " + str(epoch + 1) + \
497                          "; Train Loss: " + str(self._trainLossList[-1]) + \
498                          "; Train R2: {:5f}".format(self._train_diagnosis.R2().data) + \
499                          "; Train RMSE: {:5f}".format(self._train_diagnosis.RMSE().data) + \
500                          "; Train AIC: {:5f}".format(self._train_diagnosis.AIC()) + \
501                          "; Train AICc: {:5f}".format(self._train_diagnosis.AICc()) + \
502                          "; Valid Loss: " + str(self._validLossList[-1]) + \
503                          "; Valid R2: " + str(self._valid_r2) + \
504                          "; Learning Rate: " + str(self._optimizer.param_groups[0]['lr'])
505                logging.info(log_str)
506                if 0 < early_stop < self._noUpdateEpoch:  # stop when the model has not been updated for long time
507                    print("Training stop! Model has not been improved for over {} epochs.".format(early_stop))
508                    break
509        torch.save(self._model, self._modelSavePath + '/' + self._modelName + "_last.pkl")
510        if model_selection == "val":
511            self.load_model(self._modelSavePath + '/' + self._modelName + ".pkl")
512        elif model_selection == "last":
513            self.load_model(self._modelSavePath + '/' + self._modelName + "_last.pkl")
514        self.result_data = self.getCoefs()
